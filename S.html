class item{
	int ino;
	String iname;
}

class SI extends item{
	SI(int x,String y){
		ino=x;
		iname=y;
	}
	
	void display(){
		System.out.println(ino);
		System.out.println(iname);
	}
}


class S{
	public static void main(String args[]){
		
		SI s1=new SI(1,"book");
		s1.display();
		
		SI s2=new SI(2,"pen");
		s2.display();
		
		
	}
}
