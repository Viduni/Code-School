interface Printable{   
	 void print();   
	 }    
	 
interface Showable{   
	 void show();   
	 }   
	 
 class A implements Printable,Showable{
	 
	 public void print(){
		 System.out.print("Hello ");
		 }   
	 public void show(){
		 System.out.print("world");
		 }    
	 public static void main(String args[]){   
	 A obj = new A();   
	 obj.print();   
	 obj.show();   
	}   
}
