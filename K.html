class student{
	int sno;
	String sname;
	
	void setData(int x,String y){
		sno=x;
		sname=y;
	}
	
	void display(){
		System.out.println(sno);
		System.out.println(sname);
	}
}

class K{
	public static void main(String args[]){
		
		student a=new student();
		a.setData(1,"sunil");
		a.display();
		
		student b=new student();
		b.setData(2,"nimal");
		b.display();
		
	}
}
