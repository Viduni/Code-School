class bill{
	int price,qnt;
	
	void setBill(int x,int y){
		price=x;
		qnt=y;
	}
	
	int finBill(){
		int amt=qnt*price;
		return amt;
	}
	
	void display(int p){
		System.out.println(p);
	}
}

class L{
	public static void main(String args[]){
		
		bill a=new bill();
		a.setBill(2,5);
		int p=a.finBill();
		a.display(p);
		
		
	}
}
