//super key word

class books{
	int id;
	String name;
	books(int a,String b){
		id=a;
		name=b;
	}
}

class StoryBook extends books{
	int price;
	
	StoryBook(int x,String y,int z){
		super(x,y);
		price=z;
	}
	
	void display(){
		System.out.println(id);
		System.out.println(name);
		System.out.println(price);
	}
}

class U{
	public static void main(String args[]){
		
		StoryBook c=new StoryBook(1,"Maths",350);
		c.display();
		
		
	}
}
