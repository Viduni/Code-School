class bill{
	int price,qnt;
	
	bill(int x,int y){
		price=x;
		qnt=y;
	}
	
	int finBill(){
		int amt=qnt*price;
		return amt;
	}
	
	void display(int p){
		
		System.out.println();
		System.out.println(p);
	}
}

class M{
	public static void main(String args[]){
		
		bill a=new bill(2,5);
		//a.setBill(2,5);
		int p=a.finBill();
		a.display(p);
		
		
	}
}
